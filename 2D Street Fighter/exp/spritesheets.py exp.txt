import pygame
from os.path import isfile, join
from os import listdir



class Spritesheet:

	def __init__(self, fighter_1_steps):
		super().__init__()
		# number of frames in each row, from "fighter_1_steps" in main.py
		self.fighter_1_steps = fighter_1_steps
		self.kid_goku_spritesheets = {} #We want this dictionary to be a global variable, not a local created inside a method/ function
		
		
		
		
		
		
	#function to flip sprite
	def sprite_flip(spritesheet_frames):
		return [pygame.transform.flip(sprite, True, False) for sprite in spritesheet_frames]
		

	#function to load in images
	def load_spritesheets(self, dir1, dynamic_frame = False):

		

		#loading the right files with right directories (file destinations)
		file_path = join("assets", dir1)
		#list of all files with specified directory
		image_files = [f for f in listdir(file_path) if isfile(join(file_path, f))]

		
		
		#load images
		for image in image_files:
			for x in self.fighter_1_steps:
				sprite_sheet = pygame.image.load(join(file_path, image)).convert_alpha() 
				frame_width = sprite_sheet.get_width() // x
				#work out single frame width + height
				frame_height = sprite_sheet.get_height() / x
				

				spritesheet_frames = []


				for i in range(sprite_sheet.get_width() // frame_width):
					sprite_surface = pygame.Surface([frame_width, frame_height])
					rect_frame = pygame.Rect(i * frame_width, 100, frame_width, frame_height) #going to move across frame_width every time an image is extracted from spritesheet, rect_frame = single frame 
					sprite_surface.blit(sprite_sheet, (0,0), rect_frame) #draw spritesheet at coordinate 0,0 and it is only going to draw a portion of the sprite_sheet which is the single frame
					spritesheet_frames.append(pygame.transform.scale2x(sprite_surface))
					#print(spritesheet_frames)

				#add list of frames onto a dictionary
				if dynamic_frame == True:
					self.kid_goku_spritesheets[image.replace(".png", "") + "_right"] = spritesheet_frames
					self.kid_goku_spritesheets[image.replace(".png", "") + "_left"] = sprite_flip(spritesheet_frames)
				elif dynamic_frame == False:
					self.kid_goku_spritesheets[image.replace(".png", "")] = spritesheet_frames
				#if player turning then flip all frames the other direction
				
			#print(self.kid_goku_spritesheets)
